@override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text("Link Directory"),
        actions: [
          IconButton(
            icon: const Icon(Icons.add),
            onPressed: () {
              Navigator.of(
                context,
              ).push(MaterialPageRoute(builder: (context) => AddLink()));
            },
          ),
        ],
      ),
      body: Column(
        children: [
          Expanded(
            child: ListView.builder(
              itemCount: links.length,
              itemBuilder: (context, index) {
                return Card(
                  margin: const EdgeInsets.all(8),
                  child: ListTile(
                    title: Text(links[index]),
                    leading: const Icon(Icons.link),
                    trailing: const Icon(Icons.arrow_forward),
                    onTap: () {
                      debugPrint("Clicked on: ${links[index]}");
                    },
                  ),
                );
              },
            ),
          ),
        ],
      ),
    );
  }
}
